Service /api/maths - Documentation

GET: Maths service

RETURN: JSON object

User manual:



? op = + & x = number & y = number

return {"op":"+", "x":number, "y":number, "value": x + y}

? op = - & x = number & y = number

return {"op":"-", "x":number, "y":number, "value": x - y}

? op = * & x = number & y = number

return {"op":"*", "x":number, "y":number, "value": x * y}

? op = / & x = number & y = number

return {"op":"/", "x":number, "y":number, "value": x / y}

? op = % & x = number & y = number

return {"op":"%", "x":number, "y":number, "value": x % y}

? op = ! & n = integer

return {"op":"!", "n":integer, "value": factorial of n}

? op = p & n = integer

return {"op":"p", "n":integer, "value": true if n is a prime number}

? op = np & n = integer

return {"op":"np", "n":integer, "value": nth prime number}
